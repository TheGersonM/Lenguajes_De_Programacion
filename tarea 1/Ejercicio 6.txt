double calcularPromedio(List<double> lista) {
  double suma = 0.0;
  for (double numero in lista) {
    suma += numero;
  }

  if (lista.isNotEmpty) {
    int cant =lista.length;
    return suma / cant;
  } else {
    return 0.0; 
  }
}
